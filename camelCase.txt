package main

import (
	"fmt"
	"strings"
)

func toCamelCase(Camel string) string {

	words := strings.Split(Camel, " ")

	change := strings.ToLower(words[0])

	for i := 1; i < len(words); i++ {

		change += strings.Title(strings.ToLower(words[i]))
	}
	return change
}

func main() {
	fmt.Println(toCamelCase("HI Elvis GBAa"))
}
